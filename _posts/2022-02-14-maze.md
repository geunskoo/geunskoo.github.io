## 나의 코드

> 첫 번째 시도.
>
> ```python
> from collections import deque
> 
> n,m = map(int,input().split())
> graph = []
> for i in range(n):
>     graph.append(list(map(int,input())))
> 
> #상/하/좌/우
> dx = [-1,1,0,0] 
> dy = [0,0,-1,1]
> 
> def bfs(x,y):
>     
>     queue = deque([(x,y)])
>     while queue:
>         x,y = queue.popleft()
>            for i in range(4):
>                nx = x + dx[i]
>                ny = y + dy[i]
>                
>             if nx < 0 or ny < 0 or nx > n-1 or ny > m-1:
>                 continue
>             if graph[nx][ny] == 0:
>                    continue
>                if graph[nx][ny] == 1:
>                    graph[nx][ny] = graph[x][y]+1
>                    queue.append((nx,ny))
>     return graph[n-1][m-1]
> 
>     print(bfs(0,0))
> ```
> 
> ***>>input***
> 
> ```python
>     5 6
>    101010
>    111111
>    000001
>    111111
>     111111
>    ```
>    
>    ***>>output***
>    
>    ```python
>    10
>    ```
